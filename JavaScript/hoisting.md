# 호이스팅 - hoisting

- **익명함수가 아닌 함수**는 **스크립트가 로딩**될 때, 첫줄로 끌어올라가 **우선적으로 초기화**된다.

```jsx
// 자바스크립트는 실행시 모든 변수가 선언된다.

f2();   // f2함수가 저장된 메모리 주소

// runtime시에 해석되고 실행 (해당 루틴 실행시 이름을 가짐)
const f = function(){
    alert("익명함수");
}

f();

alert('test2')

// 스크립트가 로딩되는 시점에 바로 초기화하고 이를 VO(Variable Object)에 저장
function f2() {
    alert("선언함수");
}

/*
    선언함수
    익명함수
    test2
*/
```
